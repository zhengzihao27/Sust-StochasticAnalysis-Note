\underline{Riemann-Stieltjes Integral}: If $f$ is continuous and $g$ is of \underline{finite variation} then $\int_{a}^{b} f(x) ~ \mathrm{d}g(x)$ is well-defined and is defined to be 
$$
    \int_{a}^{b} f(x) ~ \mathrm{d}g(x):=\lim_{n\to \infty} \sum_{i=0}^{n-1}f(\xi_{i})\left(g(x_{i+1}) - g(x_{i})\right)
$$
with $a \leq x_{0} < x_{1} < \cdots < x_{n} \leq b$ and $\xi_{i} \in [x_{i}, x_{i+1}]$.

\section{Set-up}

Given a probability space $(\Omega, \mathcal{F}, \mathsf{P})$. A stochastic process $X=(X_{t})_{t \geq 0}$ is a collection of random variables taking values in $(\mathbb{R}^{d},\mathcal{B}(\mathbb{R}^{d})).$
$$
\begin{aligned}
    X\colon [0,\infty) \times \Omega &\to \mathbb{R}^{d}\\
   (t,\omega) &\mapsto X_{t}(\omega).
\end{aligned}
$$

Fix $\omega$, then the function $t \mapsto X_{t}(\omega)$ is called a sample path.

\defnn{For two stochastic process $X$ and $Y$, we say $Y$ is a modification of $X$ if $$\forall~ t \geq 0,\quad \mathsf{P}(X_{t}= Y_{t}) = 1$$}

\defnn{We say $X$ and $Y$ are indistinguishable if 
$$\mathsf{P}(X_{t} = Y_{t},~ \forall~ t \geq 0) = 1.$$ 
And then $Y$ is a version of $X$.}

\ex{}{Let $Y$ be a modification of $X$ and suppose $X$ and $Y$ both have a.s. right continuous sample paths. Then $X$ \& $Y$ are indistinguishable.}

\defnn{A function f is right-continuous if $\displaystyle f(x)= f(x+) = \lim_{t\downarrow x} f(t)$. }

\pf{Since $\forall~  t \geq 0$, $\mathsf{P}(X_{t} \neq Y_{t}) = 0,$ we get
    $$
        \mathsf{P}\left(\bigcup_{q\in \mathbb{Q}^{+}} X_{q} \neq Y_{q}\right) \leq \sum_{q\in \mathbb{Q}^{+}} \mathsf{P} (X_{q} \neq Y_{q}) = 0.
    $$
    $$
        \Rightarrow~ \mathsf{P}\left(\bigcap_{q\in \mathbb{Q}^{+}} X_{q} = Y_{q}\right) = 1.
    $$
    For two right-continuous functions $f$ and $g$, if $f(q) = g(q)$, $\forall~ q \in \mathbb{Q}^{+}$, then $f(t) = g(t)$, $\forall~ t \geq 0.$

    Hence $\mathsf{P}\left(X_{t} = Y_{t},~ \forall~ t \geq 0\right) = 1.$
}

\ex{}{Let $T \sim \mathrm{Exp}(1)$. Define 
    $$
    \begin{aligned}
        &X_{t} \equiv 0,\quad\forall~ t \geq 0 \\
        &Y_{t} = \begin{cases}
            0,& t \neq T\\
            1,&t = T.
        \end{cases}
    \end{aligned}
    $$
    $\forall~  t \geq 0$, $\mathsf{P}(X_{t} \neq Y_{t}) = \mathsf{P}(Y_{t} = 1) = \mathsf{P}(t = T) = 0.$
}

\defnn{A filtration $\{\mathcal{F}_{t}\}_{t \geq 0}$ is said to satisfy the \underline{usual condition} if 
    \begin{enumerate}[label=(\alph*)]
        \item $\displaystyle \mathcal{F}_{t} = \mathcal{F}_{t+} = \bigcap_{s>t} \mathcal{F}_{s}$;
        \item $\mathcal{F}_{0}$ contains all the \underline{$\mathsf{P}-$negligible} sets in $\mathcal{F}$.
    \end{enumerate}
}

\defnn{An even A is $\mathsf{P}$-negligible if $\exists~N \in \mathcal{F}$ s.t. $A \subseteq N$, $\mathsf{P}(N) = 0$.}

\defnn{
    $\mathcal{F}_{t+} = \bigcap_{s >t}\mathcal{F}_{s}$, $\mathcal{F}_{t-}= \sigma\left(\bigcup_{s<t}\mathcal{F}_{s}\right)$. We say a filtration is right-continuous if $\mathcal{F}_{t} = \mathcal{F}_{t+}.$
}

\defnn{
    The stochastic process $X$ is \underline{adapted} to the filtration $\{\mathcal{F}_{t}\}_{t \geq 0}$ if $\forall~t \geq 0$, $X_{t} \in \mathcal{F}_{t}$.
}

Obviously $(X_{t})$ is adapted to the natural filtration.

\defnn{The stochastic process $X$ is called progressively measurable with respect to $(\mathcal{F}_{t})$ if $\forall~t \geq 0$, the map
    $$
        \begin{aligned}
            (s,\omega) &\mapsto X_{s}(\omega)\\
            ([0,t]\times \Omega,\mathcal{B}([0,t])\otimes \mathcal{F}_{t}) & \to (\mathbb{R}^{d}, \mathcal{B}(\mathbb{R}^{d}))
        \end{aligned}
    $$
    is measurable.
}

\rmk{adapted $\supseteq$ progressively measurable $\supseteq$ previsible
}



\propp{If $X$ is adapted to the filtration ($\mathcal{F}_t$) and every sample path is right-continuous or left-continuous, then $X$ is also progressively measurable.}{
    Assume right-continuity $\forall~ t > 0$, $\forall~  n \geq 1$, $\forall~  0 \leq k \leq 2^n -1,$
    $$
    X_s^{(n)}(\omega) = X_{\frac{k+1}{2^n}t}(\omega),\quad \forall~  \frac{kt}{2^n} < s \leq \frac{k+1}{2^n}t
    $$
    Then $\displaystyle \lim_{n \to \infty}X_{s}^{n}(\omega) = X_{s}(\omega)$ for all $(s,\omega) \in [0,t]\times \Omega.$

    It suffices to prove 
    $$
    \begin{aligned}
        (s,\omega) &\mapsto X_{s}^{(n)}(\omega)\\
        ([0,t]\times \Omega,\mathcal{B}([0,t])\otimes \mathcal{F}_{t}) & \to (\mathbb{R}^{d}, \mathcal{B}(\mathbb{R}^{d}))
    \end{aligned}
    $$
    is measurable.

    $\forall~ A \in \mathcal{B}(\mathbb{R}^{d})$,
    $$
    \begin{aligned}
        &\left\{(s,\omega)\colon X_{s}^{(n)}(\omega) \in A\right\}\\
        =&\bigcup_{k=0}^{2^{n} -1}\left(\frac{k}{2^{n}}t , \frac{k+1}{2^{n}}t \right]\times \left\{X_{\frac{k+1}{2^{n}}t} \in A\right\}\in \mathcal{B}([0,t])\otimes \mathcal{F}_{t}
    \end{aligned}
    $$
    We get $X_{s}^{(n)}(\omega)$ is measurable. Hence $X_{s} = \lim_{n\to \infty}X_{s}^{(n)}$ is also measurable.
}

\section{Stopping Times}
$(\Omega, \mathcal{F},\mathsf{P})$ and $(\mathcal{F}_{t})_{t \geq 0}$.

\defnn{A random time $T$ is a \underline{stopping time} of $(\mathcal{F}_{t})$ if $\forall~ t \geq 0$, $\{T \leq t  \} \in \mathcal{F}_{t}.$
}

\defnn{A random time $T$ is an \underline{optional time} of $(\mathcal{F}_{t})$ if $\forall~ t \geq 0$, $\{T < t\} \in \mathcal{F}_{t}.$
}

\propp{stopping time $\subseteq$ optional time and stopping $=$ optional if the filtration is right-continuous.}{
    ``$\Rightarrow$'' $\{T<t\} = \bigcup_{n=1}^{\infty}\{T \leq t - \frac{1}{n}\} \in \mathcal{F}_t$.
    
    ``$\Leftarrow$'' If $T$ is an optional time, then $\forall~  m \geq 1,$
    $$
        \{T \leq t\} = \bigcap_{n=m}^{\infty}\left\{T < t+\frac{1}{n}\right\}.
    $$
    Because $\{T < t + \frac{1}{n}\} \in \mathcal{F}_{t+\frac{1}{n}} \subseteq \mathcal{F}_{t+\frac{1}{m}}$.
    
    $\Rightarrow~ \{T \leq t\} \in \mathcal{F}_{t+\frac{1}{m}}$
    
    $\Rightarrow~ \{T \leq t\} \in \bigcap_{m=1}^{\infty}\mathcal{F}_{t+\frac{1}{m}}= \mathcal{F}_{t^+} = \mathcal{F}_t$.
}

\lemnn{If $T$ and $S$ are stopping times, then $T \wedge S$, $T \vee S $, $T+S$ are stopping times.}

\pf{$\forall~ t \geq 0$, 
    \begin{enumerate}
        \item $\{T \wedge S \leq t\} = \{T \leq t\} \cup \{S \leq t\} \in \mathcal{F}_{t}.$
        \item $\{ T \vee S \leq t\} = \{T \leq t\}\cap \{S \leq t\} \in \mathcal{F}_{t}.$
        \item $\{T+S \leq t\} \in \mathcal{F}_{t} \Leftrightarrow \{T+S > t\} \in \mathcal{F}_{t}$.
        $$
        \begin{aligned}
            &\{T + S > t\} \\
            = &\{T = 0,~S >t\}\cup \{S=0,~ T> t\}\cup\{0<T<t,~T+S > t\}\cup \{S>0,~ T \geq t\}\\
             \in &\mathcal{F}_{t} \cup \mathcal{F}_{t} \cup \{0<T<t,~T+S > t\} \cup  \mathcal{F}_{t}.
        \end{aligned}
        $$
        It suffices to show
        $$\{0< T < t ,~ T+S > t\} = \bigcup_{\stackrel{r\in \mathbb{Q}^{+}}{0 < r <t}}\{r < T < t,~ S> t-r\}.
        $$
        Because the right term belong $\mathcal{F}_{t}.$
        
        ``$\Leftarrow~$'' $S > t-r > t - T$, so $S+T > t$.

        ``$\Rightarrow~$'' $\exists~ r\in (0,T)$, $S+r > t$ since $S+T > t.$
    \end{enumerate}
}

\lemnn{If $\{T_n\}_{n \geq 1}$ are stopping times, then $\sup_{n \geq 1} T_n$ is a stopping time.}

\pf{
    $\forall~ t \geq 0,$ 
    $
        \{\sup_{n \geq 1} T_n \leq t\} = \bigcap_{n=1}^{\infty}\{T_n \leq t\} \in \mathcal{F}_t.
    $
}

\defnn{The stopping $\sigma$-algebra is $\mathcal{F}_T = \{A \in \mathcal{F} \mid \forall~ t \geq 0, A \cap \{T \leq t\} \in \mathcal{F}_t\}$.}

\prop{If $T(\omega) = t_0$ for some fixed $t_0 \geq 0$, $\forall~ \omega \in \Omega$, then $\mathcal{F}_T = \mathcal{F}_{t_0}$.}

\pf{Note that $\forall~  t \geq 0$,
    $$
        \{T \leq t\} = \begin{cases}
            \varnothing,& t_0 > t\\
            \Omega, &t_0 \leq t
        \end{cases}
    $$

    \noindent``$\mathcal{F}_{t_{0}} \subseteq \mathcal{F}_{T}$''

    $\forall~  A \in \mathcal{F}_{t_{0}}$, $\forall~  t \geq 0$,
    \begin{enumerate}[label=\textcircled{\arabic*}]
        \item $t \geq t_{0}$, $A \cap \{T \leq t\} = A \in \mathcal{F}_{t_{0}} \subseteq \mathcal{F}_{t};$
        \item $t < t_{0}$, $A \cap \{T \leq t\} = \varnothing \in \mathcal{F}_{t}$.
    \end{enumerate}
    Hence $A \in \mathcal{F}_{T} \Rightarrow~ \mathcal{F}_{t_{0}}\subseteq \mathcal{F}_{T}.$

     \noindent``$\mathcal{F}_{T} \subseteq \mathcal{F}_{t_{0}}$''

    $\forall~  A \in \mathcal{F}_{T}$, we have $A \cap \{T \leq t\} \in \mathcal{F}_{t}$ by definition. Let $t = t_{0}$, then $\{T \leq t_{0}\}= \Omega$, so

    $$
        A \cap \{T \leq t_{0}\} \in \mathcal{F}_{t_{0}} \Rightarrow~ A \cap \Omega \in \mathcal{F}_{t_{0}} \Rightarrow~ A \in \mathcal{F}_{t_{0}}.
    $$
    Hence $A \in \mathcal{F}_{t_{0}} \Rightarrow~ \mathcal{F}_{T}\subseteq \mathcal{F}_{t_{0}} $
}
    
\lemnn{$\forall~$ stopping time $T$ and $S$, $\forall~A \in \mathcal{F}_S$, we have $A\cap \{S \leq T\} \in \mathcal{F}_T.$ In particular, if $S \leq T$ for all $\omega \in \Omega$, then $\mathcal{F}_S \subseteq \mathcal{F}_T.$}

\pf{
    $T \wedge t$ is also a stopping time. $\{T \wedge t \leq s\} \in \mathcal{F}_s \subseteq \mathcal{F}_t.$ Hence $T \wedge t \in \mathcal{F}_t.$

    To prove $A\cap \{S \leq T \} \in \mathcal{F}_T$, we need to prove $\forall~ t \geq 0$, $A \cap \{S \leq T\} \cap \{T \leq t\}\in \mathcal{F}_t.$

    We have 
    $$
    A \cap \{S \leq T\} \cap \{T \leq t\} = \underbrace{A \cap \{S \leq t\}}_{\in \mathcal{F}_t} \cap \underbrace{\{T \leq t\}}_{\in \mathcal{F}_t} \cap \underbrace{\{S \wedge t \leq T \wedge t\}}_{\mathcal{F}_t}.
    $$
    ($S \leq T $ on $S \leq t$, $T \leq t \Leftrightarrow \{S \wedge t \leq T \wedge t\}$)
}

\lemnn{
    Let $S,T$ be two stopping times. Then $\mathcal{F}_{T \wedge S} = \mathcal{F}_T \cap \mathcal{F}_S$ and $\{T < S\}$, $\{S < T\}$, $\{T \leq S\}$, $\{S \leq T\}$, $\{T = S\}$ belong to $\mathcal{F}_S \cap \mathcal{F}_T = \mathcal{F}_{S \wedge T}.$
}

\pf{
    $\mathcal{F}_{T \wedge S} \subseteq \mathcal{F}_S$ and $\mathcal{F}_{T \wedge S} \subseteq \mathcal{F}_T \Rightarrow~\mathcal{F}_{T \wedge S} \subseteq \mathcal{F}_T \cap \mathcal{F}_S$.
    
    To prove $\mathcal{F}_T \cap \mathcal{F}_S \subseteq \mathcal{F}_{T \wedge S}$, $\forall~ A \in \mathcal{F}_T \cap \mathcal{F}_S$, we will prove $\forall~ t \geq 0$, $A \cap \{T \wedge S \leq t\} \in \mathcal{F}_t.$

    $$
    \begin{aligned}
        A \cap \{T \wedge S \leq t\} &= A \cap \left(\{T \leq t\} \cup \{S \leq t\}\right)\\
        &= \left(A \cap \{T \leq t\}\right)\cup \left(A \cap \{S \leq t\}\right)\in \mathcal{F}_t.
    \end{aligned}
    $$

    Next $\{S \leq T\}\in \mathcal{F}_T$, $\{S > t\}\in \mathcal{F}_T$. It remains to prove $\{S <T\} \in \mathcal{F}_T.$

    Assuming this, by interchanging $T$ and $S$, we get $\{T \leq S\}$, $\{T > S\}$, $\{T <S\}\in \mathcal{F}_S$.

    Then $\{S \leq T\} \in \mathcal{F}_T$, $\{S > T\}^\complement \in \mathcal{F}_S.$

    $\Rightarrow~ \{S \leq T\} \in \mathcal{F}_S \cap \mathcal{F}_T.$

    $\{T= S\} = \{ S \leq T\}\cap \{T \leq S\}\in \mathcal{F}_S \cap \mathcal{F}_T$.

    To prove $\{S < T\}\in \mathcal{F}_T$, we define $R = S \wedge T$ such that $\{S < T\}= \{R < T\}$. Notice that $R \in \mathcal{F}_R \subseteq \mathcal{F}_T$. Hence $\{R < T\} \in \mathcal{F}_T.$
}

\textbf{Summary:}
\begin{enumerate}[label=(\roman*)]
    \item $T \in \mathcal{F}_T$;
    \item If $S \leq T$, $\mathcal{F}_S \subseteq \mathcal{ F}_T$;
    \item $T \wedge t \in \mathcal{F}_{T \wedge t} \subseteq \mathcal{F}_t$;
    \item $\mathcal{F}_{T \wedge S} = \mathcal{F}_T \cap \mathcal{F}_S.$
\end{enumerate}

\section{Cts-time martingale}
\defnn{For $(\Omega,\mathcal{F},(\mathcal{F}_{t})_{t \geq 0},\mathbb{P})$, a martingale is an adapted stochastic process $(X_t)_{t \geq 0}$ such that
    $$
    \begin{cases}
        (i)~ \forall~ t \geq 0,~\mathbb{E}|X_t| < \infty \\
        (ii)~ \forall~ t \geq s \geq 0,~\mathbb{E}(X_t \mid \mathcal{F}_s) = X _s.
    \end{cases}
    $$

    If $\mathbb{E}(X_t \mid \mathcal{F}_s) \geq X _s$, we call it a submartingale.

    If $\mathbb{E}(X_t \mid \mathcal{F}_s) \leq X _s$, we call it a supmartingale.
}

\prop{Let $\varphi: \mathbb{R} \to \mathbb{R}$ be a convex (concave up) function. If $(X_t)_{t \geq 0}$ is a martingale, then $(\varphi(X _t))_{t \geq 0}$ is a submartingale.}

\textbf{E.g.} $\abs{X_t}$, $X _{t}^2$, $\abs{X _t}^p$ for all $p \geq 1$, $(X _t)^+$ are all submartingale.

\rmk{We assume $\mathbb{E}\abs{\varphi(X_t)}<\infty$, $\forall~ t \geq 0$.}

\pf{
    Jensen's inequality.
}

\thmnn{Continuous-time Doob's upcrossing inequality}

$$
\begin{aligned}
&T_{b}^{(0)} = -\infty,\\
& T_a^{(1)} = \inf \{t \geq (T _b^{(0)})^+\colon X _t \leq a\},\\
&T_b^{(1)} = \inf \{t \geq T _a^{(1)}\colon X _t \geq b\}.
\end{aligned}
$$

$\forall~ l \geq 2$,
$$
\begin{aligned}
    &T_a^{(l)} = \inf \{t \geq T _b^{(l-1)} \colon X _t \leq a\},\\
    &T_b^{l} = \inf \{ t \geq T _a^{l}\colon X _t \geq b\}
\end{aligned}
$$

Define $U_{ab}^X [0,n] = \max \{k \geq 0\colon T_{b}^{(k)} \leq n\}$ to be the \# of upcrossing in an interval [0,n].

\thmnn{Let $(X_{t})_{t \geq 0}$ be a right-continuous submartingale. For all $T > 0$,
    $$
    \mathbb{E}\left(U_{ab}^X [0,T]\right) \leq \frac{\mathbb{E}(X_{T} -a )^+}{b-a}
    $$
}

\pf{
    For each $n \geq 1$, we let $D_n = \mathbb{Z}/ 2^n = \{0, \frac{\pm 1}{2^n}, \frac{\pm 2}{2^n},\cdots\}$.

    Consider the upcrossing of $X$ on $[0,T]\cap D_{n}$, $\left\{\frac{X_{j}}{2^n}\right\}_{j=0}^\infty$ is a discrete-time martingale. Then $U_{ab}^X\left([0,T]\cap D_{n}\right) \leq U_{ab}^X [0,T]$,
    $U_{ab}^X\left([0,T]\cap D_{n}\right) \uparrow U_{ab}^X([0,T])$.

    By using discrete-time upcrossing
    $$
        \mathbb{E}\left(U_{ab}^X\left([0,T]\cap D_{n}\right)\right) \leq \frac{1}{b-a}\mathbb{E}(X_{T_n} - a)^+,
    $$
    where $T_n = \frac{[T \cdot 2^n]}{2^n}$. Let $n\to \infty$ to get
    $$
        \mathbb{E}\left(U_{ab}^X([0,T])\right) \leq \frac{1}{b-a}\mathbb{E}(X_{T} - a)^+.
    $$  
}

\prop{
    If $(X_{t})_{t \geq 0}$ is a right continuous sub-martingale, and $\sup_{t \geq 0} \mathbb{E} X_{t}^+ < \infty$, then $\exists~ X \in L^1$ s.t. $X_t \xlongrightarrow{a.s.}X$ as $t \to \infty.$ 
}

\pf{
    $$
    \begin{aligned}
        \mathbb{E}\left(U_{ab}^X([0,\infty))\right) &= \lim_{n \to \infty} \mathbb{E}\left(U_{ab}^X([0,n])\right)\\
        &\leq \limsup_{n\to \infty}\frac{1}{b-a}\mathbb{E}(X_{n} - a)^+ \\
        &\leq \limsup_{n\to \infty}\frac{1}{b-a}\left(\mathbb{E}(X_n^+)+ \abs{a}\right) < \infty.
    \end{aligned}
    $$
    Hence $\forall~ a < b$, $U_{ab}^X[0,\infty)< \infty$ a.s. 
    
    $\Rightarrow~$w.p.1, $\forall~ a < b \in \mathbb{Q}$, $U_{ab}^X[0,\infty) < \infty$.

    Hence $\liminf_{t\to \infty}X_{t} \leq a \leq b < \limsup_{t \to \infty} X_t$ does not occur.

    $\Leftrightarrow \liminf_{t\to \infty} X_{t} = \limsup_{t \to \infty}X_t.$
}

\defnn{
    A family of random variables $(X_t)_{t \geq 0}$ is uniformly integrable (U.I.) if 
    $$
        \lim_{M \to \infty}\sup_{t \geq 0} \mathbb{E}\left(\abs{X_t} \mathsf{1}_{\abs{X _t} > M}\right) = 0.
    $$
}

\rmkb{If $\sup_{t \geq 0} \mathbb{E} \abs{X_t}^p < \infty$ for some $p > 1$, then U.I.}

\thmnn{
    If $X_n \xlongrightarrow{P} X$, then equivalent
    $$
    \begin{cases}
        X_n \xlongrightarrow{L^1} X, \\
        \mathbb{E}\abs{X_n} \to \mathbb{E}\abs{X},\\
        \{X_n\} \text{ is U.I.}.
    \end{cases}
    $$
}

\prop{Let $Z \in L^1$. Then $\{\mathbb{E}(Z\mid \mathcal{G}) \colon \mathcal{G}\text{ is a sub }\sigma \text{-field of }\mathcal{F}\}$ is U.I.

In particular $\{\mathbb{E}(Z \mid \mathcal{F}_t)\}_{t \geq 0}$ is U.I. and it is a martingale. (By Tower's property)
}

\prop{
    A right continuous martingale $(X_t)_{t \geq 0}$ is U.I. if and only if $\exists~ X _{\infty} \in L^1$ such that $X _t = \mathbb{E}(X _\infty \mid \mathcal{F}_t).$
}

\pf{
    ``$\Rightarrow$'' U.I. implies $\sup_{t \geq 0}\mathbb{E}\abs{X_t} < \infty.$

    Then the martingale convergence theorem implies that $\exists~X_{\infty}$ s.t. $X_n \xlongrightarrow{a.s.} X$ and $\mathbb{E}\abs{X _{\infty}} = \mathbb{E}\left(\lim _{t \to \infty} \abs{X _t}\right) \leq \liminf _{t \to \infty} \mathbb{E}\abs{X _t} < \infty$.

    To prove $\forall~ t \geq 0$, $X_t = \mathbb{E}(X _{\infty} \mid \mathcal{F}_t) \Leftrightarrow \forall~A \in \mathcal{F}_t$, $\mathbb{E}(X _t \mathsf{1}_A)=\mathbb{E}(X _{\infty}\mathsf{1}_A)$.

    For any $s \geq 0$, we have $\mathbb{E}(X_{t+s} \mid \mathcal{F}_{t}) = X_{t} \implies \mathbb{E}(X_{t+s}\mathsf{1}_{A}) = \mathbb{E}(X_{t}\mathsf{1}_{A})$
    $$\abs{\mathbb{E}(X_{t+s}\mathsf{1}_A) - \mathbb{E}(X_{\infty} \mathsf{ 1}_A)} \leq \mathbb{ E}\abs{X_{t+s} - X_{\infty}}\abs{\mathsf{1}_A} \leq \mathbb{E}\abs{X_{t+s}-X_{\infty}} \to 0.$$

    Let $s\to \infty$, $\mathbb{E}(X_{t+s} \mathsf{1}_{A}) \xlongrightarrow{a.s.} \mathbb{E}(X_{\infty} \mathsf{1}_{A})$.

    Recall U.I. + a.s. $\implies L^1$ convergence.

    ``$\Leftarrow$'' If $\exists~ X_\infty \in L^1$ s.t. $X_t = \mathbb{E}(X_\infty \mid \mathcal{F}_t)$, then $\{X_t\}_{t \geq 0}$ is U.I.
}

\thm{Optional sampling Theorem}{
    Let $(X_t)_{t \geq 0}$ be a right continuous martingale. Let $S \leq T$ be two stopping time. Suppose either
    \begin{enumerate}[label=\textcircled{\arabic*}]
        \item $\{X_t\}$ U.I.,
        \item $\exists~$constant $K >0$, $S,T \leq K$.
    \end{enumerate}
    Then $\mathbb{E}(X_T \mid \mathcal{F}_S) = X _S \implies \mathbb{E}(X _T) = \mathbb{E}(X _S) = \mathbb{E}(X _0).$
}

\pf{
    Define \textcircled{1} $Z = X_\infty$, \textcircled{2} $Z=X_{K}$.

    In either case, $X_t = \mathbb{E}(Z\mid \mathcal{F}_t)$, \textcircled{1} $\forall~ t \geq 0$, \textcircled{2} $\forall~ t \leq K$.

    It suffices to prove $X_T = \mathbb{E}(Z \mid \mathcal{F}_T)$. Then $X _S = \mathbb{E}(Z \mid \mathcal{F}_S)$.

    Hence 
    $$
    \begin{aligned}
    \mathbb{E}(X_T \mid \mathcal{F}_S) &= \mathbb{E}\left(\mathbb{E}(Z \mid \mathcal{F}_T) \mid \mathcal{F}_S\right)\\
    &= \mathbb{E}(Z \mid \mathcal{F}_S)\\
    &= X_S.
    \end{aligned}    
    $$

    \textcircled{1} Suppose $T \in \{t_1,t_2,\dots\}$. Then $\forall~ A \in \mathcal{F}_T$, 
    $$
    \begin{aligned}
    \mathbb{E}(X_T \mathsf{1}_A) &= \mathbb{E}(Z \mathsf{1}_A) \\
    &= \sum_{n=1}^{\infty}\mathbb{E}(X_{t_n}\mathsf{1}_{\{T=t_n\}} \mathsf{1}_A)\\
    &=\sum_{n=1}^{\infty}\mathbb{E}\left(\mathsf{1}_{A \cap \{T = t_n\}} Z\right)\\
    &= \mathbb{E}(\mathsf{1}_A Z)
    \end{aligned}$$
    because \textcircled{1} $A \cap \{T = t_n\}\in \mathcal{F}_{t_n}$, \textcircled{2} $\mathbb{E}(Z \mid \mathcal{F}_{t_n}) = X_{t_n}$

    \textcircled{2} General T. Define $T_k = \frac{[2^k T] + 1}{2^k}.$

    To prove $X_T = \mathbb{E}(Z \mid \mathcal{F}_T)$, $\forall~ A \in \mathcal{F}_T \subseteq \mathcal{F}_{T _k}$, we get $\mathbb{E}(Z \mid \mathcal{F}_{T _k}) = X _{T _k} \implies \mathbb{E}(Z \mathsf{1}_A) = \mathbb{E}(X_{T_k} \mathsf{1}_A)$.

    Send $k \to \infty$ to see $\mathbb{E}(Z \mathsf{1}_A) = \mathbb{E}(X_T \mathsf{1}_A).$

    Because \textcircled{1} $X_{T_k} \xlongrightarrow{a.s.} X_T$ by right-continuity, \textcircled{2} $\{X_{T_k}\}$ is U.I. by $X_{T_k} = \mathbb{E}(Z \mid \mathcal{F}_{T_k})$, \textcircled{3} $\implies X_{T_k} \xlongrightarrow{L^1}X_T.$
}

\thm{Doob's Maximal Inequality}{
    Let $(X_t)_{t \geq 0}$ be a right-continuous submartingale. Then $\forall~\lambda > 0,$

    $$
        \begin{aligned}
            &\mathsf{P}(\sup_{0 \leq s \leq t} X_s > \lambda) \leq \frac{1}{\lambda}\mathbb{E}X_t^+,\\
            &\mathsf{P}(\inf_{0 \leq s \leq t} X_s < -\lambda) \leq \frac{1}{\lambda}(\mathbb{E}X_t^+ - \mathbb{E}X_0).
        \end{aligned}
    $$
}

\pf{
    $\forall~ \lambda > 0$, $T = \inf \{t \geq 0: X_t > \lambda\},$ then
    $$
        \mathsf{P}(\sup_{0 \leq s \leq t}X_s > \lambda) = \mathsf{P}(T \leq t).
    $$
    
    Consider $\{X_{t \wedge T}, t \geq 0\}$,
    $$
        \begin{aligned}
        \lambda \mathsf{P}(T \leq t) = \lambda \mathbb{E}(\mathsf{1}_{T \leq t}) &\leq \mathbb{E}(X_T \mathsf{1}_{T \leq t})\\
        & = \mathbb{E}(X_{T \wedge t} \mathsf{1}_{T \leq t}) \\
        &= \mathbb{E}(X_{T \wedge t}^+ \mathsf{1}_{T \leq t})\\
        &= \mathbb{E}(X_{T \wedge t}^+) \leq \mathbb{E}(X_t^+).
        \end{aligned}
    $$

    Define $S = \inf\{t \geq 0 : X_t < -\lambda\}$, then
    $$
        \mathsf{P}(\inf_{0 \leq s \leq t}X_s < -\lambda) = \mathsf{P}(S \leq t).
    $$
    and
    $$
        \lambda \mathsf{P}(S \leq t) = \lambda \mathbb{E}(\mathsf{1}_{S \leq t}).
    $$
    
    Because $X_S \leq - \lambda$, $\lambda \leq - X_S$, so
    $$
        \lambda \mathsf{P}(S \leq t) \leq \mathbb{E}(-X_S \mathsf{1}_{S \leq t}) = - \mathbb{E}(X_{S \wedge t} \mathsf{1}_{S \leq t}).
    $$

    By submartingale optional stopping
    $$
    \begin{aligned}
    \mathbb{E}X_0 \leq \mathbb{E}X_{t \wedge S}&= \mathbb{E}(X_{t \wedge S}\mathsf{1}_{S \leq t}) + \mathbb{E}(X_{t \wedge S}\mathsf{1}_{S > t})\\
    &\leq - \lambda \mathsf{P}(S \leq t) + \mathbb{E}(X_{t \wedge S}^+) \\
    & \leq -\lambda \mathsf{P}(S \leq t)+ \mathbb{E}(X_t^+).
    \end{aligned}
    $$
}

\thmnn{$\forall~p > 1$, $\displaystyle\mathbb{E}\left(\sup_{s \leq t}\abs{X_s} \right)^p \leq \left(\frac{p}{p-1}\right)^p \mathbb{E}\abs{X_t}^p.$}

\pf{
    Define $Y= \sup_{0 \leq s \leq t}\abs{X_s}.$
    $$
    \begin{aligned}
    \lambda \mathsf{P}(Y \geq \lambda) &\leq \mathbb{E}(\abs{X_{T \wedge t}}\mathsf{1}_{T \leq t}) \\
    & = \mathbb{E}\left( \abs{X_{T \wedge t}} \right) - \mathbb{E}\left( \abs{X_{T \wedge t}} \mathsf{1}_{T > t} \right)\\
    & \leq \mathbb{E}\left( \abs{X_t} \right) - \mathbb{E}\left( \abs{X_t}\mathsf{1}_{T >t} \right)\\
    &= \mathbb{E} \abs{X_t} - \mathbb{E}\left( \abs{X_t} \mathsf{1}_{Y < \lambda} \right)\\
    &= \mathbb{E}\left( \abs{X_t}\mathsf{1}_{Y \geq \lambda} \right).
    \end{aligned}
    $$
    Hence $\mathsf{P}(Y \geq \lambda) \leq \frac{1}{\lambda}\mathbb{E}\left( \abs{X_t}\mathsf{1}_{Y \geq \lambda} \right).$
    $$
    \begin{aligned}
        \mathbb{E}Y^p &= \int_{0}^{\infty} p \lambda^{p-1}\mathsf{P}(Y \geq \lambda) ~d \lambda \\
        &\leq \int_{0}^{\infty} p \lambda^{p-2}\mathbb{E}\left( \abs{X_t}\mathsf{1}_{Y \geq \lambda} \right)~d \lambda\\
        &=\mathbb{E}\left( \abs{X_t} \int_{0}^{\infty} p \lambda^{p-2}\mathsf{1}_{Y \geq \lambda} ~d\lambda \right)\\
        &=\mathbb{E}\left( \abs{X_t}\int_{0}^{Y}p \lambda^{p-2}~d\lambda \right)\\
        &= \mathbb{E}\left(\abs{X_t}\frac{p}{p-1}Y^{p-1}\right)\\
        &\leq \frac{p}{p-1}\left( \mathbb{E}\abs{X_t}^p \right)^{\frac{1}{p}}\left( \mathbb{E}(Y^{p-1})^q \right)^{\frac{1}{q}}\\
        &=\frac{p}{p-1}\left( \mathbb{E}\abs{X_t}^p \right)^{\frac{1}{p}}\left( \mathbb{E}Y^p \right)^{\frac{1}{q}}.
    \end{aligned}
    $$
    $\implies \left( \mathbb{E}Y^p \right)^{\frac{1}{p}} \leq \frac{p}{p-1}\left( \mathbb{E}\abs{X_t}^p \right)^{\frac{1}{p}}.$

    Consider $Y \wedge k$, $\forall~ k > 0$, 
    $$
    \begin{aligned}
        \mathbb{E}\left( Y \wedge k \right)^p &= \int_{0}^{\infty} p \lambda^{p-1}\mathsf{P}(Y\wedge k \geq \lambda)~d \lambda \\
        &= \int_{0}^{k}p \lambda^{p-1}\mathsf{P}(Y \wedge k \geq \lambda)~d \lambda + \underbrace{\int_{k}^{\infty} p \lambda^{p-1}\mathsf{P}(Y\wedge k \geq \lambda)~d \lambda}_{=0} \\
        &= \int_{0}^{k}p \lambda^{p-1}\mathsf{P}(Y \wedge k \geq \lambda)~d\lambda\\
        &= \int_{0}^{k}p \lambda^{p-1}\mathsf{P}(Y \geq \lambda) ~d \lambda\\
        &\leq \mathbb{E}\left( \abs{X_t}
        \int_{0}^{k} p \lambda^{p-1} \mathsf{1}_{Y \geq \lambda}~d \lambda \right)\\
        &= \mathbb{E}\left( \abs{X_t}\int_{0}^{k \wedge Y} p \lambda^{p-2}~d \lambda \right)\\
        &= \mathbb{E}\left[ 
            \abs{X_t} \frac{p}{p-1}(Y \wedge k )^{p-1}
         \right].
    \end{aligned}
    $$

    $\implies \mathbb{E}(Y \wedge k)^p \leq \frac{p}{p-1}\left( \mathbb{E}\abs{X_t}^p \right)^{\frac{1}{p}}\left( 
        \mathbb{E}(Y \wedge k)^p
     \right)^{\frac{1}{q}}.$

    $\implies \left( \mathbb{E}(Y \wedge k)^p \right)^{\frac{1}{p}} \leq \frac{p}{p-1}\left( \mathbb{E}\abs{X_t}^p \right)^{\frac{1}{p}}$.

    Let $k\to \infty$, we get
    $$
        \left( \mathbb{E} Y^p \right)^{\frac{1}{p}} \leq \frac{p}{p-1}\left( \mathbb{E}\abs{X_t}^p \right)^{\frac{1}{p}}.
    $$
}

\subsection{The Doob-Meyer Decomposition}
Recall the Doob Decomposition
$$
    X_n = M_n + A_n,
$$
where $X_n$ is a submartingale, $M_n$ is a martingale, and $A_n$ is a predictable increasing process.

$$
    \begin{aligned}
        &\mathbb{E}\left( M_{n+1} \mid \mathcal{F}_n \right) = M_n \\
        \implies& \mathbb{E}(X_{n+1}- A_{n+1}\mid \mathcal{F}_n) = X_n - A_n \\
        \implies& \mathbb{E}(X_{n+1}\mid \mathcal{F}_n) - A_{n+1} = X_n - A_n \\
        \implies& A_{n+1} - A_n = \mathbb{E}(X_{n+1}\mid \mathcal{F}_n) - X_ n \geq 0.
    \end{aligned}
$$

$$
\begin{cases}
A_n = \sum_{k=0}^{n-1}\left( \mathbb{E}(X_{k+1} \mid \mathcal{F}_k) - X_k \right)\\
A_0 = 0.
\end{cases}
$$

Next, for a right-continuous submartingale $(X_t)_{t \geq 0}$, $X_t = M_t + A_t$, where $M_t$ is a martingale, $A_t$ is a increasing \& ``predictable''.

Consider $(A_t)_{t \geq 0}$ adapted.

\defnn{
        An adapted process $A$ is called increasing if a.s.
    \begin{enumerate}[label=(\alph*)]
        \item $A_0 = 0$;
        \item $t \mapsto A_t$ is nondecreasing and right-continuous;
        \item $\mathbb{E}(A_t) < \infty$, $\forall~ t \geq 0 $. 
    \end{enumerate}
    Integrable: $\mathbb{E}(A_\infty) = \mathbb{E}(\lim_{t \to \infty}A_t) < \infty.$
}

\defnn{
    An increasing process $A$ is called natural if $\forall~$bounded, right-continuous martingale $(M_t)_{t \geq 0}$, we have $\forall~ t > 0$,
    $$
    \mathbb{E}\int_{0}^{t} M_s ~d A_s = \mathbb{E} \int_{0}^{t} M_{s-} ~d A_s.
    $$
}
\rmkb{If $s \mapsto A_s$ is a.s. continuous, then $A$ is natural.}

\defnn{
    An increasing sequence ($A_n,\mathcal{F}_n$) is called natural if $\forall~$bounded martingale $\{M_n,\mathcal{F}_n\}$,
    $$
    \mathbb{E}(M_n A_n) = \mathbb{E}\left( \sum_{k=1}^n M_{k-1}(A_k - A_{k-1}) \right).
    $$
}

\prop{
    A predictable $\Longleftrightarrow$ A natural.
}

\pf{
    ``$\Rightarrow~$'' To prove
    $$
    \begin{aligned}
        &\mathbb{E}(M_n A_n) = \sum_{k=1}^n \mathbb{E}M_{k-1}(A_k - A_{k-1}) - \sum_{k=1}^n(M_{k-1} A_k) - \mathbb{E}(M_{k-1}A_{k-1})\\
        \Longleftrightarrow& \sum_{k=2}^n \mathbb{E}(M_{k-1}A_{k-1}) + \mathbb{E}(M_n A_n) = \sum_{k=1}^n \mathbb{E}(M_{k-1}A_k)\\
        \Longleftrightarrow &
        \sum_{k=1}^n \mathbb{E}(M_k A_k) = \sum_{k=1}^n \mathbb{E}(M_{k-1}A_k)\\
        \Longleftrightarrow &
        \sum_{k=1}^n \mathbb{E}\left[ 
            (M_k - M_{k-1})A_{k} = 0.
         \right]
    \end{aligned}
    $$

    So if $A_n \in \mathcal{F}_{n-1}$, 
    $$
    \mathbb{E}\left( 
        A_n (M_n - M_{n-1})
     \right) = \mathbb{E} \left[ 
        \underbrace{\mathbb{E}(M_n - M_{n-1} \mid \mathcal{F}_n )}_{=0}A_n 
      \right]=0.
    $$

    ``$\Leftarrow~$'' Assume $A$ is natural. Then $\forall~$bounded $(M_n)$,
    $$
    \mathbb{E}\left[ 
        A_n (M_n - M_{n-1})
     \right] = 0,\quad \forall~ n \geq 1.
    $$

    \textbf{Goal:} $A_n \in \mathcal{F}_{n-1} \Longleftrightarrow A_n = \mathbb{E}(A_n \mid \mathcal{F}_{n-1})$ a.s..
    $$
    \begin{aligned}
        &\mathbb{E}\left[ 
            M_n\left( A_n - \mathbb{E}(A_n \mid \mathcal{F}_n) \right)
         \right]\\
         =& \mathbb{E}\left( A_n M_{n-1} \right) - \mathbb{E} \left[ 
            M_n \mathbb{E}(A_n \mid \mathcal{F}_{n-1})
          \right]\\
         =& \mathbb{E}\left( M_{n-1} \left( A_n - \mathbb{E}(A_n \mid \mathcal{F}_{n-1}) \right) \right) + \mathbb{E}\left[ 
            \mathbb{E}(A_n \mid \mathcal{F}_{n-1}) (M_n - M_{n-1})
          \right]\\
         =&\mathbb{E}(M_{n-1}A_n) - \mathbb{E}\left[ \mathbb{E}(A_nM_{n-1}\mid \mathcal{F}_{n-1}) \right] + \mathbb{E}\left[ 
            \mathbb{E}(A_n \mid \mathcal{F}_{n-1}) \mathbb{E}(M_n - M_{n-1} \mid \mathcal{F}_{n-1})
          \right]= 0.
    \end{aligned}
    $$

    Fix $n \geq 1$. Let 
    $$
        X = \mathrm{sgn} \left( A_n - \mathbb{E}(A_n \mid \mathcal{F}_{n-1}) \right)
    $$
    Then $X\left( A_n - \mathbb{E}(A_n\mid \mathcal{F}_{n-1}) \right) = \abs{A_n - \mathbb{E}(A_n \mid \mathcal{F}_{n-1})}$.

    Define 
    $$ M_k = \begin{cases}
       \mathbb{E}(X \mid \mathcal{F}_{k}),& 0 \leq k \leq n-1\\
       X, &k=n
    \end{cases}$$

    Check: $\mathbb{E}(M_{k+1} \mid \mathcal{F}_{k}) = M_k$, $k \geq n$, $\mathbb{E}(X \mid \mathcal{F}_{k}) = X$.

    $$
    \begin{aligned}
        0 &= \mathbb{E}\left[ M_n (A_n - \mathbb{E}(A_n \mid \mathcal{F}_{n-1})) \right]\\
        &= \mathbb{E} \abs{A_n - \mathbb{E}(A_n \mid \mathcal{F}_{n-1})}
    \end{aligned}
    $$
    $\implies A_n = \mathbb{E}(A_n \mid \mathcal{F}_{n-1})$ a.s.
    
    $\implies A_n \in \mathcal{F}_{n-1}.$
}

$$
    \begin{aligned}
        &\mathbb{E}(M_n A_n) = \sum_{k=1}^n \mathbb{E}\left( M_{k-1} (A_k - A_{k-1}) \right) \\
        \Longleftrightarrow &
        \mathbb{E}(M_tA_t) = \mathbb{E}\left( \int_{0}^{t}M_{s-}~dA_s \right) = \mathbb{E}\left( \int_{0}^{t} M_s ~dA_s \right).
    \end{aligned}
$$

$$
    \int_{0}^{t} M_s ~d A_s = \lim_{\norm{\Pi}\to 0} \int_{0}^{t} M_s^\Pi ~d A_s.
$$
where $\Pi = \{t_0, t_1,\cdots,t_n\}\subseteq[0,t]$ with $0 = t_0 < t_1 < \cdots < t_n = t$ and $\norm{\Pi} = \max_{1 \leq k \leq n}(t_k - t_{k-1}).$ $M_s \xlongleftarrow{a.s.} M_s^\Pi = \sum_{k=1}^{n} M_{t_k}\mathsf{1}_{(t_{k-1},t_k]}(s)$ by right-continuity. For each $\Pi$, 
$$
\begin{aligned}
&\mathbb{E}\int_{0}^{t}M_s ^\Pi ~d A_s \\
=& \mathbb{E} \sum_{k=1}^n M_{t_k}(A_{t_k} - A_{t_{k-1}}) \\
=& \sum_{k=1}^n \mathbb{E}(M_{t_k A_{t_k}}) - \mathbb{E}(M_{t_k}A_{t_{k-1}})\\
=&\mathbb{E}(M_{t_n}A_{t_n}) + \sum_{k=1}^{n-1}\mathbb{E}(M_{t_{k}} A_{t_{k}}) - \sum_{k=2}^{n}\mathbb{E}(M_{t_k} A_{t_{k-1}})\\
=& \mathbb{E}(M_t A_t) + \sum_{k=1}^{n-1}\mathbb{E}(M_{t_k} A_{t_k}) - \sum_{k=1}^{n-1}\mathbb{E}(M_{t_{k+1}}A_{t_k}) \\
=& \mathbb{E}(M_t A_t) - \underbrace{\sum_{k=1}^{n}\mathbb{E}\left( A_{t_k} (M_{t_{k+1}} - M_{t_k}) \right)}_{=0}
\end{aligned}
$$

\defnn{
    Let $S = \{\text{stopping time }T \text{ with }T < \infty \text{ a.s.}\}$. We say a right-continuous $\{X_t\}_{t \geq 0}$ is of class $D$ if $\{X_T\}_{T \in S}$ is U.I. $\implies$ implies $\{X_t\}_{t \geq 0}$ is U.I..

    Fix $a > 0$. Let $S_a = \{\text{stopping time } T \text{ with }T \leq a \text{ a.s.}\}.$ $\{X_t\}_{t \geq 0}$ is of class $DL$ if $\{X_T\}_{T \in S_a}$ is U.I. for every $a > 0$.
}


\thm{Doob-Meyer Decomposition}{
    If $\{X_t\}$ is of class $DL$, then $X_t = M_t + A_t$ with $\{M_t\}$ is martingale, $\{A_t\}$ is increasing.

    If $(A_t)$ is natural, then $(M_t, A_t)$ is unique.
}

\pf{
    Assume $X_t = \begin{cases}
        M_t + A_t \\M_t' + A_t'
    \end{cases}$ with $M_t,M_t'$ martingale, $A_t, A_t'$ natural.

    Let $B_t = M_t - M_t' = A_t' - A_t$ is a martingale and of bounded variation \underline{natural}.
    
    $\forall~$bounded continuous martingale $\{\xi_t\}_{t \geq 0}$, 
    $$
    \begin{aligned}
        \mathbb{E} \xi_t (A_t - A_t')& = \mathbb{E}\int_{0}^{t}\xi_{s-} (d A_s - d A_s') \\
        &= \mathbb{E}\left[ \int_{0}^{t}\xi_{s-}  dB_s \right]\\
        &=\lim_{n \to \infty} \mathbb{E}\sum_{j=1}^{m_n}\xi_{t_{j-1}^{(n)}} \left[ B_{t_j^{(n)}} - B_{t_{j-1}^{(n)}} \right]
    \end{aligned}
    $$ 
    where $\Pi_n = \{t_0^{(n)}, \cdots, t_{m_n}^{(n)}\}$ is a partition of $[0,t]$.

    $$
    \mathbb{E}\left[ 
        \xi_{t_{j-1}^{(n)}} (B_{t_j^{(n)}} - B_{t_{j-1}^{(n)}})
     \right] = \mathbb{E} \left[ 
        \xi_{t_{j-1}^{(n)}} \underbrace{\mathbb{E}(B_{t_j^{(n)}} - B_{t_{j-1}^{(n)}} \mid \mathcal{F}_{t_{j-1}^{(n)}})}_{=0}
      \right]
    $$
    Hence $\mathbb{E}\left[ \xi_t (A_t - A_t')  \right] = 0.$

    $\forall~$bounded variable $\xi$, we let $\xi_t = \mathbb{E}(\xi \mid \mathcal{F}_t)$. Then $\mathbb{E}\left[ \mathbb{E}(\xi \mid \mathcal{F}_t)(A_t - A_t') \right] = 0 \Rightarrow~ \mathbb{E}\left[ \xi (A_t - A_t') \right] = 0$.
    
    $\xi = \mathrm{sgn}(A_t - A_t') \Rightarrow~ \mathbb{E}\abs{A_t - A_t'} = 0 \Rightarrow~A_t = A_t'$, a.s. $\forall~ t > 0 \Rightarrow~ A_t = A_t'$, $\forall~ q \in \mathbb{Q}_+$ a.s. $\Rightarrow~$Right-continuity, $A_t = A_t'$, $\forall~ t \geq 0$ a.s..

    \textbf{Existence: }Fix $a > 0$.
    
    We only need to prove $X_t = M_t + A_t$, $\forall~ 0 \leq t \leq a$.

    Consider $Y_t \triangleq X_t - \mathbb{E}(X_a \mid \mathcal{F}_t)$. $(Y_t)_{t \geq 0}$ is a submartingale because 
    $$
    \begin{aligned}
    \mathbb{E}(Y_t \mid \mathcal{F}_s)&= \mathbb{E}(X_t \mid \mathcal{F}_s) - \mathbb{E}(X_a \mid \mathcal{F}_s) \\
    &\geq X_s - \mathbb{E}(X_a \mid \mathcal{F}_s) = Y_s, \quad \forall~t \geq s.
    \end{aligned}
    $$

    $\Pi_n = \{t_0^{(n)},\cdots, t_{2^n}^{(n)}\}$ with $t_j^{(n)} = \frac{j}{2^n}a$, $0 \leq j \leq 2^n$. Then $\forall~ 0 \leq j \leq 2^n$, 
    $$
        Y_{t_{j}^{(n)}} = M_{t_{j}^{(n)}}^{(n)} + A_{t_j^{(n)}}^{(n)}
    $$
    with $A_{t_j^{(n)}}^{(n)} = \sum_{k=0}^{j-1}\mathbb{E}\left( Y_{t_{k+1}^{(n)}} - Y_{t_{k}^{(n)}} \middle| \mathcal{F}_{t_{k}^{(n)}}\right)$
    $$
    \begin{cases}
        Y_0 = X_0 - \mathbb{E}(X_a \mid \mathcal{F_0}) \leq 0\\
